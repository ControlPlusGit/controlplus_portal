143 void gen2Select(struct gen2SelectParams *p)
144 {
145 #if !RUN_ON_AS3980 && !RUN_ON_AS3981 // no select on AS3980/81 available
146  u16 len = p->mask_len;
147  u8 ebvLen;
148  u16 rxbits = 1;
149  u8 *mask = p->mask;
150  u8 j,i;
151  u8 resp_null;
152 
153  as3993ClrResponse();
154  memset(buf_,0,sizeof(buf_));
155 
156  buf_[0] = ((EPC_SELECT<<4)&0xF0) | ((p->target<<1)&0x0E) | ((p->action>>2)&0x01);
157  buf_[1] = ((p->action<<6)&0xC0) | ((p->mem_bank<<4)&0x30);
158  ebvLen = gen2InsertEBV(p->mask_address, buf_+1, 4);
159  buf_[1+ebvLen] = ((p->mask_len>>4)&0x0F);
160  buf_[2+ebvLen] = ((p->mask_len<<4)&0xF0);
161 
162  i = 2+ebvLen; /* the counter set to last buffer location */
163  for (j = len; j >= 8 ; j -= 8, mask++)
164  {
165  buf_[i] |= 0x0F & (*mask >> 4);
166  i++;
167  buf_[i] = 0xF0 & (*mask << 4);
168  }
169  buf_[i] |= ((p->truncation<<3)&0x08)/*Truncate*/;
170 
171  len += 21 + ebvLen * 8; /* Add the bits for EPC command before continuing */
172 
173  /* Pseudo 1-bit receival with small timeout to have AS3993 state machine
174  finished and avoiding spurious interrupts (no response) */
175  as3993TxRxGen2Bytes(AS3993_CMD_TRANSMCRC, buf_, len, &resp_null, &rxbits, 1, 0, 1);
176 #endif
177 }

192 {

230 /*
231  * Sends a Query, QueryRep or QueryAdjust command and checks for replies from tags.
232  * If a tag is found, the Ack command is sent. If fast mode is not enabled the Reg_RN
233  * command will be sent after receiving the Ack reply.\n
234  * All of the data which is retrieved from the tag is written into parameter tag.
235  * \param tag The reply data of the tag is written into this data structure
236  * \param qCommand The command which should be sent. Should be Query, QueryRep or QueryAdjust
237  * \param q Q as specified in the Gen2 spec
238  * \param fast If fast mode is activated no Req_RN command will be sent after receiving the
239  * Ack reply. If no QueryRep, QueryAdjust or Reg_RN is sent to the tag within T2 the tag will
240  * go back to arbitrate state.
241  * \param followCommand The command which will be sent after a slot has been processed
242  * successfully (valid tag response). Usually this will be used to send a QueryRep
243  * immediately after receiving a tag response to trigger a change of the current
244  * session flag.
245  * \return 1 if one tag has been succesfully read, 0 if no response in slot, -1 if error occured (collision)
246  */
247 static s8 gen2Slot(Tag *tag, u8 qCommand, u8 q, BOOL fast, u8 followCommand)
248 {
249  u16 rxlen;
250  s8 ret = 0;
251 
252  /*********************************************************************************/
253  /* 1. Send proper query command */
254  /*********************************************************************************/
255  rxlen = 16;
256  switch (qCommand)
257  {
258  case AS3993_CMD_QUERY:
259  gen2PrepareQueryCmd(buf_, q);
260  ret = as3993TxRxGen2Bytes(AS3993_CMD_QUERY,buf_,16,tag->rn16,&rxlen,gen2IntConfig.no_resp_time,AS3993_CMD_ACK,1);
261  break;
262  case 0: //query rep has already been sent
263  ret = as3993TxRxGen2Bytes(0,NULL,0,tag->rn16,&rxlen,gen2IntConfig.no_resp_time,AS3993_CMD_ACK,1);
264  break;
265  default:
266  ret = as3993TxRxGen2Bytes(qCommand,NULL,0,tag->rn16,&rxlen,gen2IntConfig.no_resp_time,AS3993_CMD_ACK,1);
267  }
268  if (ret == AS3993_ERR_NORES)
269  {
270  EPCLOG(" query -> noresp\n");
271  return 0;
272  }
273  if (ret < 0)
274  {
275  EPCLOG(" query -> err %hhx\n", ret);
276  return -1;
277  }
278 
279  /*********************************************************************************/
280  /* 2. Sent ACK, Receive pc, epc, send REQRN immediately after FIFO has the data */
281  /*********************************************************************************/
282  rxlen = sizeof(buf_)*8; //receive all data, length auto-calculated by AS3993
283  ret = as3993TxRxGen2Bytes(0,buf_,0,buf_,&rxlen,gen2IntConfig.no_resp_time,fast?followCommand:AS3993_CMD_REQRN,0);
284  tag->pc[0] = buf_[0];
285  tag->pc[1] = buf_[1];
286  if (ret < 0 || rxlen < 16)
287  {
288  EPCLOG(" ack rx rest(pc=%hhx) -> err %hhx\n", tag->pc[0], ret);
289  return -1;
290  }
291 #if RUN_ON_AS3980 || RUN_ON_AS3981
292  tag->epclen = (rxlen+7)/8-4; //on AS3980 crc is in the fifo, omit it.
293 #else
294  tag->epclen = (rxlen+7)/8-2;
295 #endif
296  if(tag->epclen > EPCLENGTH)
297  tag->epclen = EPCLENGTH;
298  memcpy(tag->epc, buf_+2, tag->epclen);
299 
300  if (fast)
301  {
302  gen2GetAgcRssi(&tag->agc, &tag->rssi);
303  if (((tag->pc[0] & 0xF8) >> 2) != tag->epclen)
304  {
305  return -1;
306  }
307  return 1;
308  }
309  /*********************************************************************************/
310  /* 3. Receive the two bytes handle */
311  /*********************************************************************************/
312  rxlen = 32;
313  ret = as3993TxRxGen2Bytes(0,NULL,0,tag->handle,&rxlen,gen2IntConfig.no_resp_time,followCommand,0);
314  gen2GetAgcRssi(&tag->agc, &tag->rssi);
315  if (ret < 0)
316  {
317  EPCLOG(" reqrn -> err %hhx\n", ret);
318  return -1;
319  }
320  if (((tag->pc[0] & 0xF8) >> 2) != tag->epclen)
321  {
322  return -1;
323  }
324  return 1;
325 }

423 s8 gen2AccessTag(Tag const * tag, u8 const * password)
424 {
425  s8 ret;
426  s8 error;
427  u8 count;
428  u16 rxcount;
429  u8 tagResponse[5];
430  u8 temp_rn16[2];
431 
432  for (count = 0; count < 2; count++)
433  {
434  error = gen2ReqRNHandleChar(tag->handle, temp_rn16);
435  if (error)
436  {
437  return(error);
438  }
439 
440  buf_[0] = EPC_ACCESS;
441  buf_[1] = password[0] ^ temp_rn16[0];
442  buf_[2] = password[1] ^ temp_rn16[1];
443  buf_[3] = tag->handle[0];
444  buf_[4] = tag->handle[1];
445 
446  rxcount = 32;
447  ret = as3993TxRxGen2Bytes(AS3993_CMD_TRANSMCRC,buf_,40,tagResponse,&rxcount,gen2IntConfig.no_resp_time,0,1);
448  if (ret < 0)
449  {
450  EPCLOG("access failed, tx-ret: %hhx\n", ret);
451  return GEN2_ERR_ACCESS;
452  }
453  password += 2; /* Increase pointer by two to fetch the next bytes;- */
454 
455  if ((tagResponse[1] != tag->handle[1]) ||
456  (tagResponse[0] != tag->handle[0]))
457  {
458  EPCLOG("handle not correct\n");
459  return GEN2_ERR_ACCESS;
460  }
461 #if EPCDEBUG
462  if (count ==0) EPCLOG("first part of access ok\n");
463  if (count ==1) EPCLOG("second part of access ok\n");
464 #endif
465  }
466  return ret;
467 }

504 s8 gen2KillTag(Tag const * tag, u8 const * password, u8 rfu, u8 recom, u8* tag_error)
505 {
506  s8 error;
507  u8 count;
508  u8 temp_rn16[2];
509  u16 rxbits = 32;
510  u8 cmd = AS3993_CMD_TRANSMCRC;/* first command has no header Bit */
511  u8 no_resp_time = gen2IntConfig.no_resp_time;
512 
513  *tag_error = 0xa5;
514  for (count = 0; count < 2; count++)
515  {
516  error = gen2ReqRNHandleChar(tag->handle, temp_rn16);
517  if (error)
518  {
519  break;
520  }
521 
522  if (count==1)
523  { /* Values for second part of kill */
524  cmd = AS3993_CMD_TRANSMCRCEHEAD; /* expect header bit */
525  rxbits = 32 + 1; /* add header bit */
526  rfu = recom; /* different data for rfu/recom */
527  no_resp_time = 0xff; /* waiting time up to 20ms */
528  }
529 
530  buf_[0] = EPC_KILL;
531 
532  buf_[1] = password[0] ^ temp_rn16[0];
533  buf_[2] = password[1] ^ temp_rn16[1];
534 
535  buf_[3] = ((rfu << 5) & 0xE0);
536  insertBitStream(&buf_[3], tag->handle, 2, 5);
537 
538  error = as3993TxRxGen2Bytes(cmd, buf_, 43, buf_, &rxbits, no_resp_time, 0, 1);
539 
540  if ( error )
541  {
542  break;
543  }
544  password += 2;
545  }
546  if(ERR_CHIP_HEADER == error && rxbits) *tag_error = buf_[0];
547  return error;
548 }

919  gen2IntConfig.DR = 0;

973  reg[6] = 0x05;

976  else

978  break;
979  case GEN2_LF_160: /* 160kHz */
980  reg[0] = 0x20;
981  reg[1] = 0x60;
982  if (gen2IntConfig.config.tari == TARI_125) /* Tari = 12.5us */

998  else
999  reg[7] = 0xBF; /* AS3993_REG_RXFILTER */

1001  case GEN2_LF_40: /* 40kHz */
1002  reg[0] = 0x30; /* AS3993_REG_TXOPTIONS */
1003  reg[1] = 0x00; /* AS3993_REG_RXOPTIONS */
1004  reg[2] = 0x07; /* AS3993_REG_TRCALHIGH */
1005  reg[3] = 0xD0; /* AS3993_REG_TRCALLOW */
1006  reg[4] = 0x20; /* AS3993_REG_AUTOACKTIMER */

1009  reg[7] = 0xFF; /* AS3993_REG_RXFILTER */
1010  gen2IntConfig.DR = 0;
1011  break;
1012  default:
1013  return; /* use preset settings */
1014  }
1015  reg[0] |= gen2IntConfig.config.tari;
1016  reg[1] = (reg[1] & ~0x0F)
1017  | gen2IntConfig.config.miller
1018  | (gen2IntConfig.config.trext<<3);
1019 
1020  gen2IntConfig.no_resp_time = reg[5];
1021  /* Modify only the gen2 relevant settings */
1022  as3993ContinuousWrite(AS3993_REG_TXOPTIONS, reg+0, 8);
1023 
1024  reg[0] = as3993SingleRead(AS3993_REG_TXSETTING);
1025  reg[0] = (reg[0] & ~0x03) | gen2IntConfig.config.session;
1026  as3993SingleWrite(AS3993_REG_TXSETTING, reg[0]);
1027  reg[0] = as3993SingleRead(AS3993_REG_PROTOCOLCTRL);
1028  reg[0] = reg[0] & ~0x07;
1029  as3993SingleWrite(AS3993_REG_PROTOCOLCTRL, reg[0]);
1030 }
