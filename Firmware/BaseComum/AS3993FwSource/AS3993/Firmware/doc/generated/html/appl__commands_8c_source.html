662 static u8 addToTuningTable(void)
663 {
664  u8 idx;
665  u16 iq1 = 0, iq2 = 0;
666  u32 freq;
667  freq = 0;
668  freq += (long)cmdBuffer.rxData[1];
669  freq += ((long)cmdBuffer.rxData[2]) << 8;
670  freq += ((long)cmdBuffer.rxData[3]) << 16;
671 
672  idx = tuningTable.tableSize;
673  if (idx > MAXTUNE)
674  idx = MAXTUNE;
675  tuningTable.freq[idx] = freq;
676  tuningTable.tuneEnable[idx] = 0;
677  if (cmdBuffer.rxData[4] > 0)
678  {
679  //set antenna 1
680  tuningTable.tuneEnable[idx] += 1;
681  tuningTable.cin[0][idx] = cmdBuffer.rxData[5];
682  tuningTable.clen[0][idx] = cmdBuffer.rxData[6];
683  tuningTable.cout[0][idx] = cmdBuffer.rxData[7];
684  iq1 = (u16)cmdBuffer.rxData[8];
685  iq1 += ((u16)cmdBuffer.rxData[9]) << 8;
686  tuningTable.tunedIQ[0][idx] = iq1;
687  }
688  if (cmdBuffer.rxData[10] > 0)
689  {
690  //set antenna 2
691  tuningTable.tuneEnable[idx] += 2;
692  tuningTable.cin[1][idx] = cmdBuffer.rxData[11];
693  tuningTable.clen[1][idx] = cmdBuffer.rxData[12];
694  tuningTable.cout[1][idx] = cmdBuffer.rxData[13];
695  iq2 = (u16)cmdBuffer.rxData[14];
696  iq2 += ((u16)cmdBuffer.rxData[15]) << 8;
697  tuningTable.tunedIQ[1][idx] = iq2;
698  }
699  if (tuningTable.tuneEnable[idx] > 0) //if this tuning entry is used adjust size of table.
700  tuningTable.tableSize++;
701 
702  APPLOG("add tunetable f=%x%x, tablesize=%hhx, tune1=%hhx cin1=%hhx clen1=%hhx cout1=%hhx iq1=%hx "

705  cmdBuffer.rxData[10], cmdBuffer.rxData[11], cmdBuffer.rxData[12], cmdBuffer.rxData[13], iq2);
706
