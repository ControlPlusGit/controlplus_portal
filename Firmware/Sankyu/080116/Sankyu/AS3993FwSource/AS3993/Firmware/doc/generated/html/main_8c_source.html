132 static void morse_nibble( u8 nibble )
133 {
134  u8 i, cod, code_len;
135 
136  if (nibble > 0xf) nibble = 0x89; /* -..- = "x" */
137 
138  cod = codes[nibble];
139  code_len = cod >> 5;
140 
141  for ( i = code_len; i>0; i-- )
142  {
143  if (NUMLEDS == 0) // if there are no status LEDs use the mcu led for morsing
144  MCULED(LEDON);
145  else
146  LED1(LEDON);
147  if ( cod & (1<<(i-1)) )
148  {
149  LOG("-");
150  delay_ms(450);
151  }
152  else
153  {
154  LOG(".");
155  delay_ms(150);
156  }
157  if (NUMLEDS == 0)
158  MCULED(LEDOFF);
159  else
160  LED1(LEDOFF);
161  delay_ms(150);
162  }
163  LOG(" %hhx\n", nibble);
164  delay_ms(450);
165 }

175 static void showError( u8 errorCode, u8 blinkState )
176 {
177  if (errorCode == 0)
178  {
179  MCULED(blinkState);
180  return;
181  }
182 
183  if (NUMLEDS < 2)
184  { // if we do not have enough LEDs to show error codes, morse them on LED1
185  morse_nibble(errorCode);
186  delay_ms(1000);
187  }
188  else
189  {
190  MCULED(errorCode & 0x01 ? blinkState : LEDOFF);
191  LED1(errorCode & 0x02 ? blinkState : LEDOFF);
192  LED2(errorCode & 0x04 ? blinkState : LEDOFF);
193  LED3(errorCode & 0x08 ? blinkState : LEDOFF);
194  }
195 }

275 int main(void)
276 {
277  u32 counter = 0;
278  BOOL ledBlinkState = 0;
279  u32 baudrate, realrate;
280 
281  systemInit();
282  timerInit();
283  platformInit();
284  spiInit();
285 
286  MCULED(LEDOFF);
287  LED1(LEDOFF);
288  LED2(LEDOFF);
289  LED3(LEDOFF);
290  LED4(LEDOFF);
291 
292  baudrate = BAUDRATE;
293 #if USE_UART_STREAM_DRIVER
294  uartInitialize(SYSCLK, baudrate, &realrate);
295 #else
296  uartTxInitialize(SYSCLK, baudrate, &realrate);
297 #endif
298  LOG("\nHello World\n");
299  LOG(gLogStartup, (u8)((firmwareNumber >> 16) & 0xFF), (u8)((firmwareNumber >> 8) & 0xFF), (u8)(firmwareNumber & 0xFF));
300 
301  delay_ms(1);
302  /* Fill frequencies interval used in usb_commands.c with values */
303  /* European channels since we are developping in Europe, values can be changed using GUI */
304  Frequencies.freq[0]= 866900;
305  Frequencies.freq[1]= 865700;
306  Frequencies.freq[2]= 866300;
307  Frequencies.freq[3]= 867500;
308  Frequencies.numFreqs=4;
309 
310  delay_ms(1);
311  readerInitStatus = as3993Initialize(Frequencies.freq[0]);
312 
313  initCommands(); /* USB report commands */
314 #ifdef TUNER
315  tunerInit(&mainTuner);
316 #if RADON
317  tunerInit(&ant1Tuner);
318 #endif
319 #endif
320 
321  showError(readerInitStatus, ledBlinkState);
322 
323  LOG("initialize state: %x\n", readerInitStatus);
324 
325 #if ISO6B
326  iso6bInitialize();
327 #endif
328 
329  StreamDispatcherInit(SYSCLK);
330 #if !USE_UART_STREAM_DRIVER
331  USBDeviceInit(); //usb_device.c. Initializes USB module SFRs and firmware
332  USBDeviceAttach();
333 #endif
334 
335  while (1)
336  {
337  if (counter)
338  {
339  counter--;
340  }
341  else
342  {
343  counter = (10000ULL<<(readerInitStatus?3:0));
344  ledBlinkState = (~ledBlinkState) & 0x01;
345  showError(readerInitStatus, ledBlinkState);
346  if (readerInitStatus)
347  { /* If MCU is powered before AS3993 has power it is best to try to init again */
348  readerInitStatus = as3993Initialize(Frequencies.freq[0]);
349  }
350  }
351  ProcessIO(); /* main trigger for operation commands. */
352 
353  doCyclicInventory(); /* do cyclic inventory if necessary.*/
354 
355  }

439 u8 cmdReadReg(u16 rxSize, const u8 * rxData, u16 * txSize, u8 * txData)
440 {
441  u8 status = ERR_PARAM;
442  //LOG("cmdReadReg: cmd:%hhx, rxsize:%hhx, txsize:%hhx\n", rxData[0], rxSize, *txSize);
443  if (rxSize < READ_REG_RX_SIZE)
444  return status;
445  switch (rxData[0])
446  {
447  case 0: //read all regs
448  status = readRegisters(txSize, txData);
449  break;
450  case 1: // read one reg
451  status = readRegister(rxData[1], txSize, txData);
452  break;
453  default:
454  status = ERR_PARAM;
455  }
456  return status;
457 }
