75 s8 uartTxInitialize (u32 sysclk, u32 baudrate, u32* actbaudrate)
76 {
77  u32 br1, br2;
78  u16 breg;
79 
80  /* Disable UART for configuration */
81  UART_WRITE_REG(MODE, 0x0);
82  UART_WRITE_REG(STA, 0x0);
83 
84  /* Setup UART registers */
85  /* equation according to the datasheet:
86  (sysclk / (16 * baudrate)) - 1
87  */
88  breg = (sysclk / (16 * baudrate)) - 1;
89 
90  /* round up/down w/o using floating point maths */
91  br1 = sysclk / (16 * (breg + 1));
92  br2 = sysclk / (16 * (breg + 2));
93 
94  /* check which of the two values produce fewer error rate */
95  if ((br1 - baudrate) > (baudrate - br2))
96  {
97  UART_WRITE_REG(BRG, breg + 1);
98  if ( actbaudrate )
99  {
100  *actbaudrate = br2;
101  }
102  }
103  else
104  {
105  UART_WRITE_REG(BRG, breg);
106  if ( actbaudrate )
107  {
108  *actbaudrate = br1;
109  }
110  }
111  /* Enable UART */
112  UART_WRITE_REG(MODE, UART_READ_REG(MODE) | 0x8000);
113  UART_WRITE_REG(STA, UART_READ_REG(STA) | 0x0400);
114 
115  return ERR_NONE;
116 }

200 s8 uartRxNBytes ( u8 * buffer, u16 * numBytes )
201 {
202  u16 maxCopy = *numBytes;
203  DisableIntU1RX;
204  if ( rxCount <= maxCopy )
205  { /* copy whole received data into buffer */
206  *numBytes = rxCount;
207  memcpy( buffer, rxBuffer, rxCount );
208  rxCount = 0; /* reset receive buffer */
209  }
210  else /* copy only the first *numBytes to buffer */
211  {
212  memcpy( buffer, rxBuffer, maxCopy );
213  rxCount -= maxCopy; /* remove the copied bytes from internal buffer */
214  memmove( rxBuffer, rxBuffer + maxCopy, rxCount );
215  }
216  EnableIntU1RX;
217  return ERR_NONE;
218 }
