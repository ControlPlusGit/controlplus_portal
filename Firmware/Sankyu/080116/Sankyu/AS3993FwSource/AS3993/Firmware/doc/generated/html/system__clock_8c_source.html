79 #if !defined(__PIC24FJ256GB110__) /* TODO: adapt this for GB110 (has no PLLEN but PLLDIS in config words) */
80 static void systemClockFindPllPrescaler ( u32 frequency )
81 {
82  u32 border = PLL_OUT_NOMINAL / 2 ;
83  u16 divider;
84  if ( frequency >= PLL_OUT_NOMINAL )
85  {
86  frequency = PLL_OUT_NOMINAL;
87  }
88  for ( divider = 0; divider < 3; divider++ )
89  {
90  u32 offset = border / 2;
91  if ( frequency > border + offset )
92  {
93  break; /* found it */
94  }
95  border /= 2;
96  }
97  /* either found it or stopped by 3 (which is smallest possible divider) */
98  CLKDIVbits.CPDIV = divider;
99  CLKDIVbits.PLLEN = 1; /* enable pll */
100  /* wait for lock of pll */
101  while ( ! OSCCONbits.LOCK );
102 }

111 static void systemClockFindFrcPostscaler ( u32 frequency )
112 {
113  u32 border = FRC_NOMINAL / 2 ;
114  u16 divider;
115  if ( frequency >= FRC_NOMINAL )
116  {
117  frequency = FRC_NOMINAL;
118  }
119  for ( divider = 0; divider < 7; divider++ )
120  {
121  u32 offset = border / 2;
122  if ( frequency > border + offset )
123  {
124  break; /* found it */
125  }
126  border /= 2;
127  }
128  /* either found it or stopped by 7 (which is smallest possible divider) */
129  CLKDIVbits.RCDIV = divider;
130 }

145 u32 systemClockGetFcy ( )
146 {
147  u32 frequency = 0;
148  switch ( OSCCONbits.COSC )
149  {
150 
151  case CLK_SRC_LPRC:
152  frequency = LPRC_NOMINAL;
153  break;
154 
155  case CLK_SRC_SOSC:
156  frequency = SOSC_NOMINAL;
157  break;
158 
159  case CLK_SRC_POSC_PLL:
160  if ( POSC_NOMINAL )
161  {
162  frequency = systemClockFromPllPrescaler( );
163  }
164  break;
165 
166  case CLK_SRC_POSC:
167  frequency = POSC_NOMINAL;
168  break;
169 
170  case CLK_SRC_FRCDIV:
171  frequency = systemClockFromFrcPostscaler( );
172  break;
173 
174  case CLK_SRC_FRCPLL:
175  frequency = systemClockFromPllPrescaler( );
176  break;
177 
178  case CLK_SRC_FRC:
179  frequency = FRC_NOMINAL;
180  break;
181  }
182  theSystemClock = frequency / 2; /* the theSystemClock is always half the system clock frequency */
183  return theSystemClock;
184 }
